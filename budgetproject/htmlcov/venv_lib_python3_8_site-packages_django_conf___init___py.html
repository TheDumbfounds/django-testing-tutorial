48class LazySettings(LazyObject): 

49 """ 

50 A lazy proxy for either global Django settings or a custom settings object. 

51 The user can manually configure settings prior to using them. Otherwise, 

52 Django uses the settings module pointed to by DJANGO_SETTINGS_MODULE. 

53 """ 

54 def _setup(self, name=None): 

55 """ 

56 Load the settings module pointed to by the environment variable. This 

57 is used the first time settings are needed, if the user hasn't 

58 configured settings manually. 

59 """ 

60 settings_module = os.environ.get(ENVIRONMENT_VARIABLE) 

61 if not settings_module: 

62 desc = ("setting %s" % name) if name else "settings" 

63 raise ImproperlyConfigured( 

64 "Requested %s, but settings are not configured. " 

65 "You must either define the environment variable %s " 

66 "or call settings.configure() before accessing settings." 

67 % (desc, ENVIRONMENT_VARIABLE)) 

68 

69 self._wrapped = Settings(settings_module) 

70 

71 def __repr__(self): 

72 # Hardcode the class name as otherwise it yields 'Settings'. 

73 if self._wrapped is empty: 

74 return '<LazySettings [Unevaluated]>' 

75 return '<LazySettings "%(settings_module)s">' % { 

76 'settings_module': self._wrapped.SETTINGS_MODULE, 

77 }

79 def __getattr__(self, name): 

80 """Return the value of a setting and cache it in self.__dict__.""" 

81 if self._wrapped is empty: 

82 self._setup(name) 

83 val = getattr(self._wrapped, name) 

84 

85 # Special case some settings which require further modification. 

86 # This is done here for performance reasons so the modified value is cached. 

87 if name in {'MEDIA_URL', 'STATIC_URL'} and val is not None: 

88 val = self._add_script_prefix(val) 

89 elif name == 'SECRET_KEY' and not val: 

90 raise ImproperlyConfigured("The SECRET_KEY setting must not be empty.") 

91 

92 self.__dict__[name] = val 

93 return val

95 def __setattr__(self, name, value): 

96 """ 

97 Set the value of setting. Clear all cached values if _wrapped changes 

98 (@override_settings does this) or clear single values when set.

101 self.__dict__.clear()

103 self.__dict__.pop(name, None)

107 """Delete a setting and clear it from cache if needed.""" 

108 super().__delattr__(name)

111 def configure(self, default_settings=global_settings, **options): 

112 """ 

113 Called to manually configure the settings. The 'default_settings' 

114 parameter sets where to retrieve any unspecified values from (its 

115 argument must support attribute access (__getattr__)). 

116 """ 

117 if self._wrapped is not empty: 

118 raise RuntimeError('Settings already configured.') 

119 holder = UserSettingsHolder(default_settings) 

120 for name, value in options.items(): 

121 if not name.isupper(): 

122 raise TypeError('Setting %r must be uppercase.' % name) 

123 setattr(holder, name, value) 

124 self._wrapped = holder

128 """ 

129 Add SCRIPT_NAME prefix to relative paths. 

130 

131 Useful when the app is being served at a subpath and manually prefixing 

132 subpath to STATIC_URL and MEDIA_URL in settings is inconvenient.

134 # Don't apply prefix to absolute paths and URLs. 

135 if value.startswith(('http://', 'https://', '/')): 

136 return value

138 return '%s%s' % (get_script_prefix(), value)

222class UserSettingsHolder: 

223 """Holder for user configured settings.""" 

224 # SETTINGS_MODULE doesn't make much sense in the manually configured 

225 # (standalone) case. 

226 SETTINGS_MODULE = None

228 def __init__(self, default_settings): 

229 """ 

230 Requests for configuration variables not in this class are satisfied 

231 from the module specified in default_settings (if possible).

233 self.__dict__['_deleted'] = set()

236 def __getattr__(self, name):
