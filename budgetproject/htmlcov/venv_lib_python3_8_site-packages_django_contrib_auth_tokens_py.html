72 def _make_token_with_timestamp(self, user, timestamp, legacy=False): 

73 # timestamp is number of seconds since 2001-1-1. Converted to base 36, 

74 # this gives us a 6 digit string until about 2069. 

75 ts_b36 = int_to_base36(timestamp) 

76 hash_string = salted_hmac( 

77 self.key_salt, 

78 self._make_hash_value(user, timestamp), 

79 secret=self.secret,

83 algorithm='sha1' if legacy else self.algorithm, 

84 ).hexdigest()[::2] # Limit to shorten the URL. 

85 return "%s-%s" % (ts_b36, hash_string)

87 def _make_hash_value(self, user, timestamp): 

88 """ 

89 Hash the user's primary key, email (if available), and some user state 

90 that's sure to change after a password reset to produce a token that is 

91 invalidated when it's used: 

92 1. The password field will change upon a password reset (even if the 

93 same password is chosen, due to password salting). 

94 2. The last_login field will usually be updated very shortly after 

95 a password reset. 

96 Failing those things, settings.PASSWORD_RESET_TIMEOUT eventually 

97 invalidates the token. 

98 

99 Running this data through salted_hmac() prevents password cracking 

100 attempts using the reset token, provided the secret isn't compromised. 

101 """ 

102 # Truncate microseconds so that tokens are consistent even if the 

103 # database doesn't support microseconds. 

104 login_timestamp = '' if user.last_login is None else user.last_login.replace(microsecond=0, tzinfo=None) 

105 email_field = user.get_email_field_name() 

106 email = getattr(user, email_field, '') or '' 

107 return f'{user.pk}{user.password}{login_timestamp}{timestamp}{email}'
