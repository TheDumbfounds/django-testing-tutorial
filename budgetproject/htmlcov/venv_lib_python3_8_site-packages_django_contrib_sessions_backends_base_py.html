137 # RemovedInDjango40Warning.

139 hash = self._hash(serialized)

142 def _legacy_decode(self, session_data): 

143 # RemovedInDjango40Warning: pre-Django 3.1 format will be invalid. 

144 encoded_data = base64.b64decode(session_data.encode('ascii')) 

145 try: 

146 # could produce ValueError if there is no ':' 

147 hash, serialized = encoded_data.split(b':', 1) 

148 expected_hash = self._hash(serialized) 

149 if not constant_time_compare(hash.decode(), expected_hash): 

150 raise SuspiciousSession("Session data corrupted") 

151 else: 

152 return self.serializer().loads(serialized) 

153 except Exception as e: 

154 # ValueError, SuspiciousOperation, unpickling exceptions. If any of 

155 # these happen, just return an empty dictionary (an empty session). 

156 if isinstance(e, SuspiciousOperation): 

157 logger = logging.getLogger('django.security.%s' % e.__class__.__name__) 

158 logger.warning(str(e)) 

159 return {}

246 def get_expiry_age(self, **kwargs): 

247 """Get the number of seconds until the session expires. 

248 

249 Optionally, this function accepts `modification` and `expiry` keyword 

250 arguments specifying the modification and expiry of the session. 

251 """ 

252 try: 

253 modification = kwargs['modification'] 

254 except KeyError: 

255 modification = timezone.now() 

256 # Make the difference between "expiry=None passed in kwargs" and 

257 # "expiry not passed in kwargs", in order to guarantee not to trigger 

258 # self.load() when expiry is provided. 

259 try: 

260 expiry = kwargs['expiry'] 

261 except KeyError:

267 return expiry
