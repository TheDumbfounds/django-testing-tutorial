18def get_version(version=None): 

19 """Return a PEP 440-compliant version number from VERSION.""" 

20 version = get_complete_version(version)

22 # Now build the two parts of the version number: 

23 # main = X.Y[.Z] 

24 # sub = .devN - for pre-alpha releases 

25 # | {a|b|rc}N - for alpha, beta, and rc releases 

26 

27 main = get_main_version(version) 

28 

29 sub = '' 

30 if version[3] == 'alpha' and version[4] == 0: 

31 git_changeset = get_git_changeset() 

32 if git_changeset: 

33 sub = '.dev%s' % git_changeset 

34 

35 elif version[3] != 'final': 

36 mapping = {'alpha': 'a', 'beta': 'b', 'rc': 'rc'} 

37 sub = mapping[version[3]] + str(version[4]) 

38 

39 return main + sub

42def get_main_version(version=None): 

43 """Return main version (X.Y[.Z]) from VERSION.""" 

44 version = get_complete_version(version) 

45 parts = 2 if version[2] == 0 else 3 

46 return '.'.join(str(x) for x in version[:parts])

49def get_complete_version(version=None):

72def get_git_changeset(): 

73 """Return a numeric identifier of the latest git changeset. 

74 

75 The result is the UTC timestamp of the changeset in YYYYMMDDHHMMSS format. 

76 This value isn't guaranteed to be unique, but collisions are very unlikely, 

77 so it's sufficient for generating the development version numbers. 

78 """ 

79 repo_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__))) 

80 git_log = subprocess.run( 

81 ['git', 'log', '--pretty=format:%ct', '--quiet', '-1', 'HEAD'], 

82 stdout=subprocess.PIPE, stderr=subprocess.PIPE, 

83 shell=True, cwd=repo_dir, universal_newlines=True, 

84 ) 

85 timestamp = git_log.stdout 

86 try: 

87 timestamp = datetime.datetime.utcfromtimestamp(int(timestamp)) 

88 except ValueError: 

89 return None 

90 return timestamp.strftime('%Y%m%d%H%M%S')

93def get_version_tuple(version): 

94 """ 

95 Return a tuple of version numbers (e.g. (1, 2, 3)) from the version 

96 string (e.g. '1.2.3'). 

97 """ 

98 loose_version = LooseVersion(version) 

99 version_numbers = [] 

100 for item in loose_version.version: 

101 if not isinstance(item, int): 

102 break 

103 version_numbers.append(item) 

104 return tuple(version_numbers)
