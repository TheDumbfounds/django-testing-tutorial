219 def random_elements(self, elements=('a', 'b', 'c'), length=None, unique=False, 

220 use_weighting=None): 

221 """Generate a list of randomly sampled objects from ``elements``. 

222 

223 Set ``unique`` to ``False`` for random sampling with replacement, and set ``unique`` to 

224 ``True`` for random sampling without replacement. 

225 

226 If ``length`` is set to ``None`` or is omitted, ``length`` will be set to a random 

227 integer from 1 to the size of ``elements``. 

228 

229 The value of ``length`` cannot be greater than the number of objects 

230 in ``elements`` if ``unique`` is set to ``True``. 

231 

232 The value of ``elements`` can be any sequence type (``list``, ``tuple``, ``set``, 

233 ``string``, etc) or an ``OrderedDict`` type. If it is the latter, the keys will be 

234 used as the objects for sampling, and the values will be used as weighted probabilities 

235 if ``unique`` is set to ``False``. For example: 

236 

237 .. code-block:: python 

238 

239 # Random sampling with replacement 

240 fake.random_elements( 

241 elements=OrderedDict([ 

242 ("variable_1", 0.5), # Generates "variable_1" 50% of the time 

243 ("variable_2", 0.2), # Generates "variable_2" 20% of the time 

244 ("variable_3", 0.2), # Generates "variable_3" 20% of the time 

245 ("variable_4": 0.1), # Generates "variable_4" 10% of the time 

246 ]), unique=False 

247 ) 

248 

249 # Random sampling without replacement (defaults to uniform distribution) 

250 fake.random_elements( 

251 elements=OrderedDict([ 

252 ("variable_1", 0.5), 

253 ("variable_2", 0.2), 

254 ("variable_3", 0.2), 

255 ("variable_4": 0.1), 

256 ]), unique=True 

257 ) 

258 

259 :sample: elements=('a', 'b', 'c', 'd'), unique=False 

260 :sample: elements=('a', 'b', 'c', 'd'), unique=True 

261 :sample: elements=('a', 'b', 'c', 'd'), length=10, unique=False 

262 :sample: elements=('a', 'b', 'c', 'd'), length=4, unique=True 

263 :sample: elements=OrderedDict([ 

264 ("a", 0.45), 

265 ("b", 0.35), 

266 ("c", 0.15), 

267 ("d", 0.05), 

268 ]), length=20, unique=False 

269 :sample: elements=OrderedDict([ 

270 ("a", 0.45), 

271 ("b", 0.35), 

272 ("c", 0.15), 

273 ("d", 0.05), 

274 ]), unique=True 

275 """ 

276 use_weighting = (use_weighting 

277 if use_weighting is not None 

278 else self.__use_weighting__) 

279 

280 if isinstance(elements, dict) and not isinstance(elements, OrderedDict): 

281 raise ValueError("Use OrderedDict only to avoid dependency on PYTHONHASHSEED (See #363).") 

282 

283 fn = choices_distribution_unique if unique else choices_distribution 

284 

285 if length is None: 

286 length = self.generator.random.randint(1, len(elements)) 

287 

288 if unique and length > len(elements): 

289 raise ValueError( 

290 "Sample length cannot be longer than the number of unique elements to pick from.") 

291 

292 if isinstance(elements, dict): 

293 if not hasattr(elements, "_key_cache"): 

294 elements._key_cache = tuple(elements.keys()) 

295 

296 choices = elements._key_cache 

297 probabilities = tuple(elements.values()) if use_weighting else None 

298 else: 

299 if unique: 

300 # shortcut 

301 return self.generator.random.sample(elements, length) 

302 choices = elements 

303 probabilities = None 

304 

305 return fn( 

306 tuple(choices), 

307 probabilities, 

308 self.generator.random, 

309 length=length,

367 def randomize_nb_elements( 

368 self, 

369 number=10, 

370 le=False, 

371 ge=False, 

372 min=None, 

373 max=None): 

374 """Generate a random integer near ``number`` according to the following rules: 

375 

376 - If ``le`` is ``False`` (default), allow generation up to 140% of ``number``. 

377 If ``True``, upper bound generation is capped at 100%. 

378 - If ``ge`` is ``False`` (default), allow generation down to 60% of ``number``. 

379 If ``True``, lower bound generation is capped at 100%. 

380 - If a numerical value for ``min`` is provided, generated values less than ``min`` 

381 will be clamped at ``min``. 

382 - If a numerical value for ``max`` is provided, generated values greater than 

383 ``max`` will be clamped at ``max``. 

384 - If both ``le`` and ``ge`` are ``True``, the value of ``number`` will automatically 

385 be returned, regardless of the values supplied for ``min`` and ``max``. 

386 

387 :sample: number=100 

388 :sample: number=100, ge=True 

389 :sample: number=100, ge=True, min=120 

390 :sample: number=100, le=True 

391 :sample: number=100, le=True, max=80 

392 :sample: number=79, le=True, ge=True, min=80 

393 """ 

394 if le and ge: 

395 return number 

396 _min = 100 if ge else 60 

397 _max = 100 if le else 140

399 if min is not None and nb < min: 

400 nb = min 

401 if max is not None and nb > max: 

402 nb = max 

403 return nb

405 def numerify(self, text='###'): 

406 """Generate a string with each placeholder in ``text`` replaced according 

407 to the following rules: 

408 

409 - Number signs ('#') are replaced with a random digit (0 to 9). 

410 - Percent signs ('%') are replaced with a random non-zero digit (1 to 9). 

411 - Exclamation marks ('!') are replaced with a random digit or an empty string. 

412 - At symbols ('@') are replaced with a random non-zero digit or an empty string. 

413 

414 Under the hood, this method uses :meth:`random_digit() <faker.providers.BaseProvider.random_digit>`, 

415 :meth:`random_digit_not_null() <faker.providers.BaseProvider.random_digit_not_null>`, 

416 :meth:`random_digit_or_empty() <faker.providers.BaseProvider.random_digit_or_empty>`, 

417 and :meth:`random_digit_not_null_or_empty() <faker.providers.BaseProvider.random_digit_not_null_or_empty>` 

418 to generate the random values. 

419 

420 :sample: text='Intel Core i%-%%##K vs AMD Ryzen % %%##X' 

421 :sample: text='!!! !!@ !@! !@@ @!! @!@ @@! @@@' 

422 """ 

423 text = _re_hash.sub( 

424 lambda x: str(self.random_digit()), 

425 text) 

426 text = _re_perc.sub( 

427 lambda x: str(self.random_digit_not_null()), 

428 text) 

429 text = _re_excl.sub( 

430 lambda x: str(self.random_digit_or_empty()), 

431 text) 

432 text = _re_at.sub( 

433 lambda x: str(self.random_digit_not_null_or_empty()), 

434 text) 

435 return text
