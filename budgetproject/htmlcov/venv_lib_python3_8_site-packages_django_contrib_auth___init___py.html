41@sensitive_variables('credentials') 

42def _clean_credentials(credentials): 

43 """ 

44 Clean a dictionary of credentials of potentially sensitive info before 

45 sending to less secure functions. 

46 

47 Not comprehensive - intended for user_login_failed signal 

48 """ 

49 SENSITIVE_CREDENTIALS = re.compile('api|token|key|secret|password|signature', re.I) 

50 CLEANSED_SUBSTITUTE = '********************' 

51 for key in credentials: 

52 if SENSITIVE_CREDENTIALS.search(key): 

53 credentials[key] = CLEANSED_SUBSTITUTE 

54 return credentials

63@sensitive_variables('credentials') 

64def authenticate(request=None, **credentials): 

65 """ 

66 If the given credentials are valid, return a User object. 

67 """ 

68 for backend, backend_path in _get_backends(return_tuples=True): 

69 backend_signature = inspect.signature(backend.authenticate) 

70 try: 

71 backend_signature.bind(request, **credentials) 

72 except TypeError: 

73 # This backend doesn't accept these credentials as arguments. Try the next one. 

74 continue 

75 try: 

76 user = backend.authenticate(request, **credentials) 

77 except PermissionDenied: 

78 # This backend says to stop in our tracks - this user should not be allowed in at all. 

79 break 

80 if user is None: 

81 continue 

82 # Annotate the user object with the path of the backend. 

83 user.backend = backend_path 

84 return user 

85 

86 # The credentials supplied are invalid to all backends, fire signal 

87 user_login_failed.send(sender=__name__, credentials=_clean_credentials(credentials), request=request)

90def login(request, user, backend=None): 

91 """ 

92 Persist a user id and a backend in the request. This way a user doesn't 

93 have to reauthenticate on every request. Note that data set during 

94 the anonymous session is retained when the user logs in. 

95 """ 

96 session_auth_hash = '' 

97 if user is None: 

98 user = request.user 

99 if hasattr(user, 'get_session_auth_hash'): 

100 session_auth_hash = user.get_session_auth_hash() 

101 

102 if SESSION_KEY in request.session: 

103 if _get_user_session_key(request) != user.pk or ( 

104 session_auth_hash and 

105 not constant_time_compare(request.session.get(HASH_SESSION_KEY, ''), session_auth_hash)): 

106 # To avoid reusing another user's session, create a new, empty 

107 # session if the existing session corresponds to a different 

108 # authenticated user. 

109 request.session.flush() 

110 else: 

111 request.session.cycle_key() 

112 

113 try: 

114 backend = backend or user.backend 

115 except AttributeError: 

116 backends = _get_backends(return_tuples=True) 

117 if len(backends) == 1: 

118 _, backend = backends[0] 

119 else: 

120 raise ValueError( 

121 'You have multiple authentication backends configured and ' 

122 'therefore must provide the `backend` argument or set the ' 

123 '`backend` attribute on the user.' 

124 ) 

125 else: 

126 if not isinstance(backend, str): 

127 raise TypeError('backend must be a dotted import path string (got %r).' % backend) 

128 

129 request.session[SESSION_KEY] = user._meta.pk.value_to_string(user) 

130 request.session[BACKEND_SESSION_KEY] = backend 

131 request.session[HASH_SESSION_KEY] = session_auth_hash 

132 if hasattr(request, 'user'): 

133 request.user = user 

134 rotate_token(request) 

135 user_logged_in.send(sender=user.__class__, request=request, user=user)

141 data. 

142 """ 

143 # Dispatch the signal before the user is logged out so the receivers have a

147 user = None

149 request.session.flush()

151 from django.contrib.auth.models import AnonymousUser

169def get_user(request): 

170 """ 

171 Return the user model instance associated with the given request session. 

172 If no user is retrieved, return an instance of `AnonymousUser`. 

173 """ 

174 from .models import AnonymousUser 

175 user = None 

176 try: 

177 user_id = _get_user_session_key(request) 

178 backend_path = request.session[BACKEND_SESSION_KEY] 

179 except KeyError: 

180 pass 

181 else: 

182 if backend_path in settings.AUTHENTICATION_BACKENDS: 

183 backend = load_backend(backend_path) 

184 user = backend.get_user(user_id) 

185 # Verify the session 

186 if hasattr(user, 'get_session_auth_hash'): 

187 session_hash = request.session.get(HASH_SESSION_KEY) 

188 session_hash_verified = session_hash and constant_time_compare( 

189 session_hash, 

190 user.get_session_auth_hash() 

191 ) 

192 if not session_hash_verified: 

193 if not ( 

194 session_hash and 

195 hasattr(user, '_legacy_get_session_auth_hash') and 

196 constant_time_compare(session_hash, user._legacy_get_session_auth_hash()) 

197 ): 

198 request.session.flush() 

199 user = None 

200 

201 return user or AnonymousUser()
